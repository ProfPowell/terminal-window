React Integration

Installation

Install terminal-window via npm:

npm install terminal-window
Basic Usage

Import and use in your React component:

import { useEffect, useRef } from 'react';
import 'terminal-window';

function App() {
  const terminalRef = useRef(null);

  useEffect(() => {
    const terminal = terminalRef.current;
    terminal.registerCommand('react', () => {
      return 'Hello from React!';
    });
  }, []);

  return (
    <terminal-window
      ref={terminalRef}
      theme="dark"
      prompt="$ "
      autofocus
    />
  );
}
Wrapper Component

Create a reusable wrapper with event handling:

const Terminal = forwardRef(({
  onCommand, commands, ...props
}, ref) => {
  const terminalRef = useRef(null);

  useImperativeHandle(ref, () => ({
    executeCommand: (cmd) =>
      terminalRef.current?.executeCommand(cmd),
    print: (text, type) =>
      terminalRef.current?.print(text, type),
    clear: () =>
      terminalRef.current?.clear(),
  }));

  useEffect(() => {
    const terminal = terminalRef.current;
    if (commands) {
      Object.entries(commands).forEach(
        ([name, handler]) => {
          terminal.registerCommand(name, handler);
        }
      );
    }
  }, [commands]);

  return <terminal-window
    ref={terminalRef} {...props} />;
});
TypeScript Types

Add type declarations for JSX:

// types/terminal-window.d.ts
declare namespace JSX {
  interface IntrinsicElements {
    'terminal-window': React.DetailedHTMLProps<
      React.HTMLAttributes<HTMLElement> & {
        theme?: 'dark' | 'light';
        prompt?: string;
        'enable-vfs'?: boolean;
        autofocus?: boolean;
      },
      HTMLElement
    >;
  }
}
Next.js / App Router

Use dynamic imports for SSR:

'use client';

import { useEffect, useRef } from 'react';

export default function ClientTerminal(props) {
  const terminalRef = useRef(null);

  useEffect(() => {
    import('terminal-window');
  }, []);

  return <terminal-window
    ref={terminalRef} {...props} />;
}
