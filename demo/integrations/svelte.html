Svelte Integration

Installation

Install terminal-window via npm:

npm install terminal-window
No config needed! Svelte has excellent native web component support.
Basic Usage

Works out of the box:

<script>
  import { onMount } from 'svelte';
  import 'terminal-window';

  let terminal;

  onMount(() => {
    terminal.registerCommand('svelte', () => {
      return 'Hello from Svelte!';
    });
  });

  function handleCommand(event) {
    console.log('Command:', event.detail);
  }
</script>

<terminal-window
  bind:this={terminal}
  theme="dark"
  prompt="svelte> "
  on:command={handleCommand}
  autofocus
/>
Svelte 5 (Runes)

Using the new runes syntax:

<script>
  import { onMount } from 'svelte';
  import 'terminal-window';

  let terminal = $state(null);
  let history = $state([]);

  onMount(() => {
    terminal.registerCommand('count', () => {
      return `Commands: ${history.length}`;
    });
  });

  function handleCommand(event) {
    history = [...history, event.detail];
  }
</script>

<terminal-window
  bind:this={terminal}
  theme="dark"
  oncommand={handleCommand}
  autofocus
/>
SvelteKit

Use dynamic imports for SSR:

<script>
  import { onMount } from 'svelte';
  import { browser } from '$app/environment';

  let terminal;

  onMount(async () => {
    if (browser) {
      await import('terminal-window');
    }
  });
</script>

{#if browser}
  <terminal-window
    bind:this={terminal}
    theme="dark"
    autofocus
  />
{/if}
Wrapper Component

Create a reusable component:

<!-- lib/Terminal.svelte -->
<script>
  import { onMount, createEventDispatcher } from 'svelte';
  import 'terminal-window';

  export let theme = 'dark';
  export let prompt = '$ ';
  export let commands = {};

  let terminal;
  const dispatch = createEventDispatcher();

  onMount(() => {
    Object.entries(commands).forEach(
      ([name, handler]) => {
        terminal.registerCommand(name, handler);
      }
    );
  });

  export function executeCommand(cmd) {
    terminal?.executeCommand(cmd);
  }
</script>

<terminal-window
  bind:this={terminal}
  {theme}
  {prompt}
  on:command={(e) => dispatch('command', e.detail)}
  autofocus
/>
