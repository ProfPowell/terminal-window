11ty Integration

Installation

Install terminal-window via npm:

npm install terminal-window
Configuration

Set up passthrough copy in .eleventy.js:

// .eleventy.js
module.exports = function(eleventyConfig) {
  // Copy terminal-window dist files
  eleventyConfig.addPassthroughCopy({
    'node_modules/terminal-window/dist':
      'js/terminal-window'
  });

  return {
    dir: {
      input: 'src',
      output: '_site'
    }
  };
};
Basic Usage

Use in your templates:

<!-- _includes/layouts/base.njk -->
<!DOCTYPE html>
<html>
<head>
  <script type="module"
    src="/js/terminal-window/terminal-window.es.js">
  </script>
</head>
<body>
  {{ content | safe }}
</body>
</html>
In Markdown

Embed directly in markdown files:

---
layout: layouts/base.njk
title: My Tutorial
---

# Welcome

Here's an interactive terminal:

<terminal-window
  theme="dark"
  prompt="$ "
  enable-vfs
  autofocus
></terminal-window>
Shortcode

Create a reusable shortcode:

// .eleventy.js
eleventyConfig.addShortcode('terminal', (options = {}) => {
  const {
    theme = 'dark',
    prompt = '$ ',
    enableVfs = false,
    id = 'terminal'
  } = options;

  return `
    <terminal-window
      id="${id}"
      theme="${theme}"
      prompt="${prompt}"
      ${enableVfs ? 'enable-vfs' : ''}
      autofocus
    ></terminal-window>
  `;
});

Then use in templates:

{% terminal theme="dark", prompt="$ ", enableVfs=true %}
With Commands

Add custom commands via front matter:

---
layout: layouts/tutorial.njk
title: Git Basics
terminalCommands:
  status: "On branch main..."
  log: "commit abc1234..."
---

# Git Tutorial

{% terminal id="git-terminal" %}

<script>
const commands = {{ terminalCommands | dump | safe }};
const terminal = document.getElementById('git-terminal');
Object.entries(commands).forEach(([name, response]) => {
  terminal.registerCommand(name, () => response);
});
</script>
